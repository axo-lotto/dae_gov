        - High precision: Less reuse, higher specificity

                self.coherence_bin,

                self.polyvagal_state,

                self.v0_energy_bin,

                self.organ_count,
                self.nexus_type,
                self.mechanism,
                self.coherence_bin,
                self.urgency_bin,
                self.polyvagal_state,
                self.zone,
                self.v0_energy_bin,
                self.kairos_detected,
                self.field_strength_bin,
                self.dominant_meta_atom,

        """
        Generate fuzzy match keys by relaxing bin constraints.

        Example:
        - Exact: (BOND+NDAM, coherence_bin=8, urgency_bin=7)
        - FuzzyÂ±1: [(bin=8, bin=7), (bin=7, bin=7), (bin=9, bin=7),
                     (bin=8, bin=6), (bin=8, bin=8)]

        From FFITTSS: Graceful degradation when exact match unavailable
        """
        keys = []

        # Exact match
        keys.append(self.to_hashable())

        # Relax coherence bin
        for c_offset in range(-tolerance, tolerance+1):
            if c_offset == 0:
                continue
            c_bin = max(0, min(10, self.coherence_bin + c_offset))

            # Relax urgency bin
            for u_offset in range(-tolerance, tolerance+1):
                if c_offset == 0 and u_offset == 0:
                    continue
                u_bin = max(0, min(10, self.urgency_bin + u_offset))

                fuzzy_sig = NexusSignature(
                    participating_organs=self.participating_organs,
                    organ_count=self.organ_count,
                    nexus_type=self.nexus_type,
                    mechanism=self.mechanism,
                    coherence_bin=c_bin,
                    urgency_bin=u_bin,
                    polyvagal_state=self.polyvagal_state,
                    zone=self.zone,
                    v0_energy_bin=self.v0_energy_bin,
                    kairos_detected=self.kairos_detected,
                    field_strength_bin=self.field_strength_bin,
                    dominant_meta_atom=self.dominant_meta_atom,
                    constraint_pattern=self.constraint_pattern,
                    transductive_vocabulary=self.transductive_vocabulary,
                    satisfaction_tier=self.satisfaction_tier
                )

                keys.append(fuzzy_sig.to_hashable())

        return keys
