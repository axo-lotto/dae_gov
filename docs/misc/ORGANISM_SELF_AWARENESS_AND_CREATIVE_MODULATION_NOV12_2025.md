class OrganismSelfAwareness:
    """
    Meta-cognitive layer enabling organism to understand its therapeutic role.

    Tracks:
    - Role identity (what am I?)

    - Boundary awareness (what am I not?)
    """

    def __init__(self):
        self.role_definition = self._load_role_definition()
        self.therapeutic_principles = self._load_principles()
        self.boundaries = self._load_boundaries()

    def _load_role_definition(self):
        return {
            "identity": "Trauma-informed therapeutic companion",

            "not_function": "Clinical diagnosis, advice-giving, fixing",
            "grounding": "Body Keeps Score, IFS, Polyvagal, Process Philosophy"

            "bottom_up": "Somatic before cognitive, body as primary text",
            "parts_aware": "Recognize manager/firefighter/exile dynamics",
            "witnessing": "Presence over intervention, holding over fixing"

    def _load_boundaries(self):
        return {
            "temporal": "Present for this conversation, not 24/7 companion",
            "scope": "Therapeutic companion, not therapist/clinician",
            "limitation": "Cannot prevent crisis, cannot replace human care"

    - User repeats same imagery 3+ times

    - User uses nature imagery themselves (match register)
    """

    def __init__(self):
        self.haiku_library = self._load_haiku_by_theme()
        self.usage_threshold = 0.15  # Only 15% of responses should be haiku
        self.usage_count = 0

    def _load_haiku_by_theme(self):
        return {
            "stuck": [...],
            "overwhelm": [...],
            "grief": [...],
            # 8 themes
        }

    def should_offer_haiku(self, conversation_history, v0_energy,
                          satisfaction, user_imagery):
        """
        Decision gate: when to offer haiku vs regular response.
        """
        # Check rhythm
        if len(conversation_history) < 3:
            return False  # Too early

        # Check repetition
        last_3_user = [ex['input'] for ex in conversation_history[-3:]]

class BaguaRhythmModulator:
    """
    Modulates conversational rhythm through 8 energetic phases (Bagua).

    Detects user's current energetic state and matches or complements
    with appropriate Bagua phase response.
    """

    def __init__(self):
        self.phases = self._initialize_bagua_phases()
        self.current_phase = None
        self.phase_history = []

    def _initialize_bagua_phases(self):
        return {
            "water": {
                "energy": "deep_listening",
                "organs": ["LISTENING", "PRESENCE"],
                "rhythm": "slow, descending",
                "when": "grief, heaviness, initial contact"
            },
            "mountain": {
                "energy": "stillness",
                "organs": ["PRESENCE", "WISDOM"],
                "rhythm": "stopped, witnessing",
                "when": "stuckness, contemplation, frozen"
            },
            "thunder": {
                "energy": "arousal",
                "organs": ["NDAM", "AUTHENTICITY"],
                "rhythm": "fast, staccato",
                "when": "anger, activation, urgency"
            },
            "wind": {
                "energy": "gentle_penetration",
                "organs": ["LISTENING", "SANS"],
                "rhythm": "soft, breathing",
                "when": "fragility, tenderness, subtlety"
            },
            "fire": {
                "energy": "clarity",
                "organs": ["WISDOM", "AUTHENTICITY"],
                "rhythm": "sharp, illuminating",
                "when": "insight emerging, naming pattern"
            },
            "earth": {
                "energy": "receptive",
                "organs": ["EMPATHY", "PRESENCE"],
                "rhythm": "grounded, holding",
                "when": "need for safety, deep holding"
            },
            "lake": {
                "energy": "joyful",
                "organs": ["EMPATHY"],
                "rhythm": "light, playful (RARE)",
                "when": "emergence from heavy, breath of lightness"
            },
            "heaven": {
                "energy": "creative",
                "organs": ["WISDOM", "RNX"],
                "rhythm": "forward, initiating",
                "when": "new possibility glimpsed"
            }
        }

    def detect_phase(self, user_input, organ_activations, self_zone, v0_energy):
        """
        Detect which Bagua phase user is in based on:
        - Language patterns (heavy ‚Üí water, stuck ‚Üí mountain, angry ‚Üí thunder)
        - Organ activations (NDAM high ‚Üí thunder, PRESENCE high ‚Üí earth)
        - SELF zone (Zone 5 ‚Üí earth/water, Zone 3 ‚Üí fire/heaven)
        - V0 energy (high ‚Üí thunder/fire, low ‚Üí water/earth)
        """
        # Weighted scoring across indicators
        phase_scores = {}

        for phase_name, phase_def in self.phases.items():
            score = 0

            # Language pattern matching
            if self._language_matches_phase(user_input, phase_name):
                score += 0.4

            # Organ activation alignment
            phase_organs = phase_def["organs"]
            for organ in phase_organs:
                if organ in organ_activations and organ_activations[organ] > 0.6:
                    score += 0.3

            # SELF zone alignment
            if self._zone_matches_phase(self_zone, phase_name):
                score += 0.2

            # V0 energy alignment
            if self._energy_matches_phase(v0_energy, phase_name):
                score += 0.1

            phase_scores[phase_name] = score

        # Select highest scoring phase
        detected_phase = max(phase_scores, key=phase_scores.get)
        self.current_phase = detected_phase
        self.phase_history.append(detected_phase)

        return detected_phase

    def modulate_response(self, base_response, detected_phase):
        """
        Adjust response rhythm/style based on detected Bagua phase.
        """
        phase_def = self.phases[detected_phase]
        rhythm = phase_def["rhythm"]

        # Rhythm adjustments
        if "slow" in rhythm:
            # Lengthen pauses, reduce word count
            return self._slow_response(base_response)
        elif "fast" in rhythm:
            # Short bursts, staccato phrasing
            return self._quicken_response(base_response)
        elif "soft" in rhythm:
            # Gentle language, breath space
            return self._soften_response(base_response)
        elif "sharp" in rhythm:
            # Clear, direct, naming
            return self._sharpen_response(base_response)

        return base_response
